import numpy as np
import matplotlib.pyplot as plt


def main():
    x = np.linspace(0, 20, 100)
    y = np.exp(1.1**x)

    plt.figure()
    plt.plot(x, y)
    plt.xlabel('$x$')
    plt.ylabel('$\exp(x)$')
    plt.title('e^k*x')
    
    plt.figure()
    plt.plot(x, -np.exp(-x))
    plt.xlabel('$x$')
    plt.ylabel('$-\exp(-x)$')

    plt.show()
    
if __name__ == '__main__':
    main()
