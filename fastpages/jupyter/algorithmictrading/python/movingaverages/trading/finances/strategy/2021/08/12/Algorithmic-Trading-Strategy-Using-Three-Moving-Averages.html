Algorithmic Trading Strategy Using Three Moving Averages & Python

import os
import pandas as pd
import pandas_datareader as web
import numpy as np
import matplotlib.pyplot as plt
import datetime as dt
plt.style.use('fivethirtyeight')

# You can use a local file or you can load data directly from the web
def symbol_to_path(symbol, base_dir="data"):
    """Return CSV file path given ticker symbol."""
    return os.path.join(base_dir, "{}.csv".format(str(symbol)))

#df = pd.read_csv(symbol_to_path("TSLA"), index_col='Date', parse_dates=True, na_values=['nan']) ##, usecols=['Date', 'Adj Close']

#Load Data from web
#Get Dates From DateEntry and Convert It To Datetime
start = dt.datetime(2019, 12, 1)
end = dt.datetime(2020, 6, 1)
    
df = web.DataReader('TSLA', 'yahoo', start, end)    

#df = df.set_index(pd.DatetimeIndex(df['Date'].values))

plt.figure(figsize=(12.2, 4.5))
plt.title('Close price', fontsize = 22)
plt.plot(df['Close'])
plt.xlabel('Date', fontsize = 18)
plt.ylabel('Close price', fontsize = 18)
plt.show()

#Calculate the short/ fast exponential moving average
ShortEMA = df.Close.ewm(span=5, adjust = False).mean()

#Calculate the Middle/ Medium exponential moving average
MiddleEMA = df.Close.ewm(span=21, adjust = False).mean()

#Calculate the long/ slow exponential moving average
LongEMA = df.Close.ewm(span=63, adjust = False).mean()

plt.figure(figsize=(12.2, 4.5))
plt.title('Close price', fontsize = 22)
plt.plot(df['Close'], label= 'Close Price', color = 'blue')
plt.plot(ShortEMA, label= 'Short/Fast', color = 'red')
plt.plot(MiddleEMA, label= 'Middle/medium', color = 'orange')
plt.plot(LongEMA, label= 'Long/Slow', color = 'green')
plt.xlabel('Date', fontsize = 18)
plt.ylabel('Close price', fontsize = 18)
plt.show()

df['Short']=ShortEMA
df['Middle']=MiddleEMA
df['Long']=LongEMA

def buy_sell_function(data):
    buy_list = []
    sell_list = []
    flag_long = False
    flag_short = False
    
    for i in range(0, len(data)):
        if data['Middle'][i] < data['Long'][i] and data['Short'][i] < data['Middle'][i] and flag_long == False and flag_short == False:
            buy_list.append(data['Close'][i])
            sell_list.append(np.nan)
            flag_short = True
        elif flag_short == True and data['Short'][i] > data['Middle'][i]:
            sell_list.append(data['Close'][i])
            buy_list.append(np.nan)
            flag_short = False
        elif data['Middle'][i] > data['Long'][i] and data['Short'][i] > data['Middle'][i] and flag_long == False and flag_short == False:
            buy_list.append(data['Close'][i])
            sell_list.append(np.nan)
            flag_long = True
        elif flag_long == True and data['Short'][i] < data['Middle'][i]:
            sell_list.append(data['Close'][i])
            buy_list.append(np.nan)
            flag_long = False
        else:
            buy_list.append(np.nan)
            sell_list.append(np.nan)
        
        
    return  (buy_list, sell_list)   
        
         

df['Buy'] = buy_sell_function(df)[0]
df['Sell'] = buy_sell_function(df)[1]

# Visualize de closing prize and the exponential moving averages

plt.figure(figsize=(12.2, 4.5))
plt.title('Close price', fontsize = 22)
plt.plot(df['Close'], label= 'Close Price', color = 'blue', alpha = 0.25)
plt.plot(ShortEMA, label= 'Short/Fast', color = 'red', alpha = 0.25)
plt.plot(MiddleEMA, label= 'Middle/medium', color = 'orange', alpha = 0.25)
plt.plot(LongEMA, label= 'Long/Slow', color = 'green', alpha = 0.25, )
plt.scatter(df.index, df['Buy'], color = 'green', marker='^', alpha = 1)
plt.scatter(df.index, df['Sell'], color = 'red', marker='v', alpha = 1)
plt.xlabel('Date', fontsize = 18)
plt.ylabel('Close price', fontsize = 18)
plt.show()
